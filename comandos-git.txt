git init // inicializa um novo repositório
git add "nome arquivo" // adiciona o arquivo atual ao próximo commit
git add .  // adiciona os arquivos atuais ao próximo commit
git status // verifica o status atual dos repositórios git
git commit -m "mensagem do commit" // cria um novo commit com uma mensagem
git push // envia as atualizações para a nuvem na branch ativa
git branch // permite listar e ver qual branch está ativa atualmente
git checkout nome-da-branch // permite mudar para uma nova branch
git checkout -b "nome da branch de origem" "nome da nova branch" // permite mudar e criar uma nova branch com base em outra
git merge "branch a receber merge" // permite fazer merge(unir) a branch ativa atualmente com outra branch
git pull // atualiza a branch atualmente ativa
touch .gitignore // cria um arquivo txt para conseguir ignorar os arquivos listados
git reflog // mostra o log de atualizações
git reset --hard *id-a-recuperar* // permite recuperar a versão anterior
